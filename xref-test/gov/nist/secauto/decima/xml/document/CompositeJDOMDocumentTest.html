51    private static final Namespace COMPOSITE_NS = Namespace.getNamespace("http://decima.nist.gov/xml/test");
52    private static final Namespace TEMPLATE_A_NS = Namespace.getNamespace("templateA");
53    private static final Namespace TEMPLATE_B_NS = Namespace.getNamespace("templateB");

55    protected CompositeXMLDocument buildComposite() throws DocumentException, MalformedURLException {
56      XMLDocument sub1 = new JDOMDocument(new URL("classpath:templates/templateA.xml"));
57      XMLDocument sub2 = new JDOMDocument(new URL("classpath:templates/templateB.xml"));
58      Map<String, XMLDocument> subs = new HashMap<>();
59      subs.put("sub1", sub1);
60      subs.put("sub2", sub2);
61  
62      return new CompositeXMLDocument(new URL("classpath:templates/composite.xml"), subs);
63    }

70    @Test
71    public void testXPathRoot()
72        throws DocumentException, MalformedURLException, XPathExpressionException, XPathFactoryConfigurationException {
73      CompositeXMLDocument base = buildComposite();
74  
75      XPathEvaluator eval = base.newXPathEvaluator();
76      // JDOMBasedXPathEvaluator eval = new JDOMBasedXPathEvaluator(base.getJDOMDocument(false));
77      XPathNamespaceContext nsContext = new XPathNamespaceContext("http://decima.nist.gov/xml/test");
78      nsContext.addNamespace("t", "http://decima.nist.gov/xml/test");
79      eval.setNamespaceContext(nsContext);
80      Element root = eval.evaluateSingle("/t:test", Filters.element());
81      Assert.assertNotNull("XPath did not match", root);
82  
83      XPathContext context = base.getContext(root);
84  
85      ContextAssert.assertContext("/*[local-name()='test' and namespace-uri()='http://decima.nist.gov/xml/test'][1]", 2,
86          96, "classpath:templates/composite.xml", context);
87    }

89    @Test
90    public void testXPathTemplateA() throws DocumentException, MalformedURLException, XPathExpressionException {
91      CompositeXMLDocument base = buildComposite();
92  
93      JDOMBasedXPathEvaluator eval = new JDOMBasedXPathEvaluator(base.getJDOMDocument(false));
94      XPathNamespaceContext nsContext = new XPathNamespaceContext("http://decima.nist.gov/xml/test");
95      nsContext.addNamespace("a", "templateA");
96      eval.setNamespaceContext(nsContext);
97      List<Element> elements = eval.evaluate("//a:templateA", Filters.element());
98      Assert.assertEquals("XPath should have 2 result elements", 2, elements.size());
99  
100     Element element = elements.get(0);
101 
102     // relative to the templateA
103     ContextAssert.assertContext("/*[local-name()='templateA' and namespace-uri()='templateA'][1]", 2, 30,
104         "classpath:templates/templateA.xml", base.getContext(element));
105 
106     element = elements.get(1);
107 
108     // relative to the templateA
109     ContextAssert.assertContext("/*[local-name()='templateA' and namespace-uri()='templateA'][1]", 2, 30,
110         "classpath:templates/templateA.xml", base.getContext(element));
111   }

113   @Test
114   public void test() throws DocumentException, IOException {
115     CompositeXMLDocument base = buildComposite();
116     Document compiledDocument = base.getJDOMDocument();
117     Element root = compiledDocument.getRootElement();
118     // System.err.println(JDOMUtil.toString(root));
119     assertElement("test", COMPOSITE_NS, root);
120 
121     // Assert composite test1 contains sub1
122     List<Element> elements = root.getContent(Filters.element());
123     Element child = elements.get(0);
124     assertElement("test1", COMPOSITE_NS, child);
125     Element sub = child.getChild("templateA", TEMPLATE_A_NS);
126     assertElement("templateA", TEMPLATE_A_NS, sub);
127 
128     // Assert composite test2 contains sub2
129     child = elements.get(1);
130     assertElement("test2", COMPOSITE_NS, child);
131     sub = child.getChild("templateB", TEMPLATE_B_NS);
132     assertElement("templateB", TEMPLATE_B_NS, sub);
133 
134     // Assert composite test3 contains sub1
135     child = elements.get(2);
136     assertElement("test3", COMPOSITE_NS, child);
137     sub = child.getChild("templateA", TEMPLATE_A_NS);
138     assertElement("templateA", TEMPLATE_A_NS, sub);
139   }
140 
141   private static void assertElement(String localname, Namespace ns, Element element) {
142     org.junit.Assert.assertEquals(localname, element.getName());
143     org.junit.Assert.assertTrue(ns.equals(element.getNamespace()));
144   }
