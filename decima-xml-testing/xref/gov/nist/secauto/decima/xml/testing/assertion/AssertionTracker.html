27  package gov.nist.secauto.decima.xml.testing.assertion;
28  
29  import gov.nist.secauto.decima.core.assessment.result.BaseRequirementResult;
30  import gov.nist.secauto.decima.core.assessment.result.DerivedRequirementResult;
31  
32  import java.util.HashSet;
33  import java.util.Set;
34  
35  /**
36   * Used to track which assertions have been checked vs. which have not. This is needed to support
37   * {@link RemainingAssertion} semantics.
38   */
39  public class AssertionTracker {
40    private final Set<BaseRequirementResult> assertedBaseRequirements = new HashSet<>();
41    private final Set<DerivedRequirementResult> assertedDerivedRequirements = new HashSet<>();
42  
43    /**
44     * Registered the provided base requirement as asserted.
45     * 
46     * @param result
47     *          the base requirement result to register
48     * @throws AssertionException
49     *           if the requirement has already been asserted
50     */
51    public void assertRequirement(BaseRequirementResult result) throws AssertionException {
52      if (!assertedBaseRequirements.add(result)) {
53        throw new AssertionException(
54            "The base requirement '" + result.getBaseRequirement().getId() + "' has already been asserted.");
55      }
56    }
57  
58    /**
59     * Registered the provided derived requirement as asserted.
60     * 
61     * @param result
62     *          the derived requirement result to register
63     * @throws AssertionException
64     *           if the requirement has already been asserted
65     */
66    public void assertRequirement(DerivedRequirementResult result) throws AssertionException {
67      if (!assertedDerivedRequirements.add(result)) {
68        throw new AssertionException(
69            "The derived requirement '" + result.getDerivedRequirement().getId() + "' has already been asserted.");
70      }
71    }
72  
73    public boolean isAsserted(BaseRequirementResult result) {
74      return assertedBaseRequirements.contains(result);
75    }
76  
77    public boolean isAsserted(DerivedRequirementResult result) {
78      return assertedDerivedRequirements.contains(result);
79    }
80  }
