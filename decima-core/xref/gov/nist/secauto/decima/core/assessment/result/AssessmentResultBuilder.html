31  import gov.nist.secauto.decima.core.document.Document;
32  import gov.nist.secauto.decima.core.requirement.RequirementsManager;

34  import java.util.Map;

36  /**
37   * Implementations of this interface can be used to build an {@link AssessmentResults} based on
38   * results reported to the builder during an assessment. An {@link AssessmentResultBuilder} is used
39   * by the various {@link Assessment} implementations to track a collection of {@link TestResult}
40   * instances generated during the assessment process. new TestResult instances can be registered by
41   * calling the {@link #addTestResult(Assessment, Document, String, TestResult)} method. Once all
42   * assessments are completed, an {@link AssessmentResults} instance can be generated using the
43   * {@link #build(RequirementsManager)} method. Before calling the
44   * {@link DefaultAssessmentResultBuilder#build(RequirementsManager)}, callers must first call the
45   * {@link DefaultAssessmentResultBuilder#end()} method to designate the end of the assessment run.
46   * <p>
47   * This interface supports marking specific derived requirements as tested by calling the
48   * {@link #assignTestStatus(Assessment, Document, String, TestState)} method. This can be useful to
49   * ensure that the derived requirement {@link ResultStatus} does not get evaluated as
50   * {@link ResultStatus#NOT_TESTED} when a TestResult has not been generated for the derived
51   * requirement because there was no failure to report using the
52   * {@link #addTestResult(Assessment, Document, String, TestResult)} method.
53   */
54  public interface AssessmentResultBuilder {
55    /**
56     * Retrieves the mapping of derived requirement identifiers to {@link TestState}. To be tested the
57     * derived requirement must: 1) have an associated {@link TestResult} provided by calling the
58     * {@link #addTestResult(Assessment, Document, String, TestResult)} method, or 2) be declared as a
59     * specific {@link TestState} by calling the
60     * {@link #assignTestStatus(Assessment, Document, String, TestState)} method.
61     * 
62     * @return a map of derived requirement identifiers to the corresponding test status
63     */
64    Map<String, TestState> getTestStateByDerivedRequirementId();
65  
66    /**
67     * Calling this method signals the start of the assessment. The assessment will be automatically
68     * started the first time either {@link #addTestResult(Assessment, Document, String, TestResult)} or
69     * {@link #assignTestStatus(Assessment, Document, String, TestState)} is called. Multiple calls to
70     * this method will not change the start time from the initial time.
71     * 
72     * @return the same builder instance
73     */
74    AssessmentResultBuilder start();
75  
76    /**
77     * Calling this method signals the end of the assessment.
78     * 
79     * @return the same builder instance
80     */
81    AssessmentResultBuilder end();
82  
83    /**
84     * Allows the target of an assessment to be registered and tracked. This can be called multiple
85     * times for the same target without causing duplicate targets to be registered. It is expected that
86     * an {@link Assessment} will call this method at least once.
87     * 
88     * @param document
89     *          the target of the assessment
90     * @return the same builder instance
91     */
92    AssessmentResultBuilder addAssessmentTarget(Document document);
93  
94    /**
95     * Allows arbitrary assessment properties to be associated with the results. This can be used to
96     * include meta information in the assessment results produced.
97     * 
98     * @param key
99     *          the property key
100    * @param value
101    *          the property value
102    * @return this builder
103    */
104   AssessmentResultBuilder assignProperty(String key, String value);
105 
106   /**
107    * Appends a new {@link TestResult} to the collection of test results associated with the provided
108    * derived requirement identifier.
109    * 
110    * @param <DOC>
111    *          the type of document that is the target of the assessment
112    * @param assessment
113    *          the target assessment
114    * @param document
115    *          the document being assessed
116    * @param derivedRequirementId
117    *          the derived requirement that the test result is associated with
118    * @param result
119    *          the test result to append
120    * @return the same builder instance
121    */
122   <DOC extends Document> AssessmentResultBuilder addTestResult(Assessment<? extends DOC> assessment, DOC document,
123       String derivedRequirementId, TestResult result);
124 
125   /**
126    * Marks the derived requirement associated with the provided identifier as having a specific
127    * TestState. This method is called when the
128    * {@link #addTestResult(Assessment, Document, String, TestResult)} method is called. If no test
129    * result is reported for a given derived requirement, then this method needs to be called to
130    * indicate the actual test state of a derived requirement.
131    * 
132    * @param <DOC>
133    *          the type of document that is the target of the assessment
134    * @param assessment
135    *          the target assessment
136    * @param document
137    *          the document being assessed
138    * @param derivedRequirementId
139    *          the derived requirement to set the state for
140    * @param state
141    *          the state to assign
142    * @return the current builder instance
143    */
144   <DOC extends Document> AssessmentResultBuilder assignTestStatus(Assessment<? extends DOC> assessment, DOC document,
145       String derivedRequirementId, TestState state);
146 
147   /**
148    * Generates a new {@link AssessmentResults} instance based on the previously recorded
149    * {@link TestResult} records and other statuses provided to the builder..
150    * <p>
151    * Note: {@link #end()} must be called before this method to terminate the assessment.
152    * 
153    * @param requirementsManager
154    *          the set of requirements corresponding to the assessments performed
155    * @return a new assessment result instance
156    * @throws IllegalStateException
157    *           if the {@link #end()} method hasn't first been called
158    */
159   AssessmentResults build(RequirementsManager requirementsManager);
160 
161   /**
162    * Retrieves a handler that is used to process various assessment event callback methods.
163    * 
164    * @return a {@code non-null} handler instance
165    */
166   LoggingHandler getLoggingHandler();
167 
168   void setLoggingHandler(LoggingHandler handler);
169 }
