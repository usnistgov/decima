80    @Test
81    public void testDeleteAction() throws Exception {
82      String baseXML = getStandardBaseXML();
83      File testXmlFile = tempFolder.newFile();
84  
85      // Delete an element
86      StringBuilder actionXML = new StringBuilder();
87      actionXML.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
88      actionXML.append("<template xmlns='" + DECIMA_TEMPLATE_NS_URI + "' template='" + testXmlFile.toURI()
89          + "' xmlns:temp='http://tempuri.org'>");
90      actionXML.append("	<delete xpath='/temp:root-node/temp:sample-node-1'/>");
91      actionXML.append("</template>");
92  
93      String newXML = processActions(baseXML, testXmlFile, actionXML.toString());
94      String controlXML = baseXML.replace("<sample-node-1/>", "");
95      Diff diff = getNewDiff(controlXML, newXML);
96      if (!diff.identical()) {
97        fail(diff.toString(), controlXML, newXML);
98      }
99  
100     // Delete an attribute
101     actionXML = new StringBuilder();
102     actionXML.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
103     actionXML.append("<template xmlns='" + DECIMA_TEMPLATE_NS_URI + "' template='" + testXmlFile.toURI()
104         + "' xmlns:temp='http://tempuri.org'>");
105     actionXML.append("	<delete xpath='/temp:root-node/temp:sample-node-2/@sample'/>");
106     actionXML.append("</template>");
107 
108     newXML = processActions(baseXML, testXmlFile, actionXML.toString());
109     controlXML = baseXML.replace("sample='test'", "");
110     diff = getNewDiff(controlXML, newXML);

454   private String processActions(String baseXML, File baseFile, String actionXML)
455       throws UnsupportedEncodingException, JDOMException, IOException, TemplateParserException, DocumentException {
456     Document baseDoc = buildDocumentFromString(baseXML);
457 
458     XMLOutputter xout = new XMLOutputter();
459     xout.output(baseDoc, new FileWriter(baseFile));
460 
461     TemplateParser parser = TemplateParser.getInstance();
462     InputStream inputStream = new ByteArrayInputStream(actionXML.getBytes(Charset.forName("UTF-8")));
463     TemplateProcessor tp = parser.parse(inputStream, baseFile.toURI().toURL());
464     XMLDocument document = tp.generate(new SimpleXMLDocumentResolver());
465     return document.asString(Format.getRawFormat());

478   private Document buildDocumentFromString(String xml) throws UnsupportedEncodingException, JDOMException, IOException {
479     return builder.build(new ByteArrayInputStream(xml.toString().getBytes("UTF-8")));
480   }

482   private void fail(String message, String controlXML, String testXML) {
483     Assert.fail(message + "\n\nControl:\n " + controlXML + "\n\nResult:\n" + testXML);
484   }

486   private Diff getNewDiff(String controlXML, String testXML) throws SAXException, IOException {
487     XMLUnit.setIgnoreWhitespace(true);
488     Diff diff = new Diff(controlXML, testXML);
489     diff.overrideDifferenceListener(new CustomDifferenceListener());
490     return diff;
491   }
492 
493   private static class CustomDifferenceListener implements DifferenceListener {
494     @Override
495     public int differenceFound(Difference difference) {
496       if (difference.getDescription().equals("namespace prefix")) {
497         return DifferenceListener.RETURN_IGNORE_DIFFERENCE_NODES_IDENTICAL;
498       }
499       return DifferenceListener.RETURN_ACCEPT_DIFFERENCE;
500     }
