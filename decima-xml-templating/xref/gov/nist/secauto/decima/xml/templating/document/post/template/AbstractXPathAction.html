31  import org.apache.logging.log4j.LogManager;
32  import org.apache.logging.log4j.Logger;
33  import org.jdom2.Document;
34  import org.jdom2.filter.Filter;
35  import org.jdom2.xpath.XPathBuilder;
36  import org.jdom2.xpath.XPathExpression;
37  import org.jdom2.xpath.XPathFactory;
38  
39  import java.util.List;
40  import java.util.Map;
41  import java.util.Objects;
42  
43  /**
44   * This abstract implementation provides basic XPath query support for {@link Action}
45   * implementations that handle specific action operations.
46   * 
47   * @param <T>
48   *          The generic type of the results of the XPath query after being processed by the JDOM
49   *          {@code Filter<T>}
50   */
51  public abstract class AbstractXPathAction<T> implements XPathAction<T> {
52    private static final Logger logger = LogManager.getLogger(AbstractXPathAction.class);
53  
54    // TODO: replace this with the XMLDocument XPath functionality
55    private final XPathExpression<T> xpath;
56  
57    /**
58     * Construct a new AbstractXPathAction based on an XPath string, a JDOM {@code Filter<T>} using the
59     * provided namespace map to map XML prefixes to namespaces.
60     * 
61     * @param xpathFactory
62     *          the XPath implementation to use
63     * @param xpath
64     *          the XPath string
65     * @param filter
66     *          a filter to determine the types of nodes to return
67     * @param prefixToNamespaceMap
68     *          a map of XML prefixes to namespaces used in the provided XPath
69     */
70    public AbstractXPathAction(XPathFactory xpathFactory, String xpath, Filter<T> filter,
71        Map<String, String> prefixToNamespaceMap) {
72      Objects.requireNonNull(xpathFactory);
73      Objects.requireNonNull(xpath);
74      ObjectUtil.requireNonEmpty(xpath);
75      Objects.requireNonNull(filter);
76      Objects.requireNonNull(prefixToNamespaceMap);
77      this.xpath = buildXPath(xpathFactory, xpath, filter, prefixToNamespaceMap);
78    }
79  
80    @Override
81    public XPathExpression<T> getXpath() {
82      return xpath;
83    }
84  
85    /**
86     * Compiles the provided XPath.
87     * 
88     * @param xpath
89     *          the XPath string
90     * @param filter
91     *          a filter to determine the types of nodes to return
92     * @param prefixToNamespaceMap
93     *          a map of XML prefixes to namespaces used in the provided XPath
94     * @return a compiled JDOM XPath expression
95     */
96    private static <T> XPathExpression<T> buildXPath(XPathFactory xpathFactory, String xpath, Filter<T> filter,
97        Map<String, String> prefixToNamespaceMap) {
98      XPathBuilder<T> builder = new XPathBuilder<T>(xpath, filter);
99      for (Map.Entry<String, String> entry : prefixToNamespaceMap.entrySet()) {
100       if (!entry.getKey().isEmpty()) {
101         builder.setNamespace(entry.getKey(), entry.getValue());
102       }
103     }
104     return builder.compileWith(xpathFactory);
105   }
106 
107   /**
108    * Resolves the XPath expression against the provided document, returning the nodeset.
109    * 
110    * @param document
111    *          the document to query against
112    * @return a list of matching nodes
113    * @throws ActionException
114    *           if an error occurs while resolving the XPath expression
115    */
116   protected List<T> resolveXpath(Document document) throws ActionException {
117     List<T> results;
118     try {
119       results = getXpath().evaluate(document);
120     } catch (IllegalArgumentException e) {
121       throw new InvalidXPathActionException("Invalid XPath: " + xpath, e);
122     } catch (Throwable e) {
123       throw new InvalidXPathActionException("An unexpected exception occured while processing XPath: " + xpath, e);
124     }
125     if (results.isEmpty()) {
126       throw new NoXPathResultsActionException("No XPath results found for XPath: " + xpath);
127     }
128     return results;
129   }
130 
131   @Override
132   public void execute(Document document) throws ActionException {
133     if (logger.isTraceEnabled()) {
134       logger.trace("Evaluating XPath: {}", getXpath().getExpression());
135     }
136     List<T> results = resolveXpath(document);
137     if (logger.isTraceEnabled()) {
138       logger.trace("Processing {} XPath result(s)", results.size());
139     }
140     process(results);
141   }
142 
143   /**
144    * Perform the implemented action on the provided XPath results.
145    * 
146    * @param results
147    *          a list of XPath results guaranteed to be non-empty
148    * @throws ActionException
149    *           if an error occurs while processing the implemented action
150    */
151   protected abstract void process(List<T> results) throws ActionException;
152 }
