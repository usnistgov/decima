27  package gov.nist.secauto.decima.xml.templating.document.post.template;
28  
29  import gov.nist.secauto.decima.core.util.ObjectUtil;
30  import gov.nist.secauto.decima.xml.jdom2.NamespaceUtil;
31  
32  import org.jdom2.Element;
33  import org.jdom2.Namespace;
34  import org.jdom2.filter.Filters;
35  import org.jdom2.xpath.XPathFactory;
36  
37  import java.util.List;
38  import java.util.Map;
39  import java.util.Objects;
40  
41  /**
42   * Adds a new XML attribute to an exiting element.
43   */
44  public class AddAttributeAction extends AbstractXPathAction<Element> {
45    private final String namespace;
46    private final String name;
47    private final String value;
48  
49    /**
50     * Construct a new AddAttributeAction based on an XPath string using the provided map to map XML
51     * prefixes to namespaces.
52     * 
53     * @param xpathFactory
54     *          the XPath implementation to use
55     * @param xpath
56     *          the XPath string
57     * @param prefixToNamespaceMap
58     *          a map of XML prefixes to namespaces used in the provided XPath
59     * @param namespace
60     *          the namespace of the new attribute or {@code null} if the attribute has no namespace
61     * @param name
62     *          the name of the new attribute
63     * @param value
64     *          the value of the new attrubute
65     */
66    public AddAttributeAction(XPathFactory xpathFactory, String xpath, Map<String, String> prefixToNamespaceMap,
67        String namespace, String name, String value) {
68      super(xpathFactory, xpath, Filters.element(), prefixToNamespaceMap);
69      ObjectUtil.requireNullOrNonEmpty(namespace, "namespace must be null or non-empty");
70      Objects.requireNonNull(name);
71      Objects.requireNonNull(value);
72      this.namespace = namespace;
73      this.name = name;
74      this.value = value;
75    }
76  
77    /**
78     * Retrieve the namespace URI to use for the new attribute.
79     * 
80     * @return the namespace URI
81     */
82    public String getNamespace() {
83      return namespace;
84    }
85  
86    /**
87     * Retrieve the name of the new attribute.
88     * 
89     * @return the attribute's name
90     */
91    public String getName() {
92      return name;
93    }
94  
95    /**
96     * Retrieve the new attribute's value.
97     * 
98     * @return the new attributes value
99     */
100   public String getValue() {
101     return value;
102   }
103 
104   /**
105    * Appends an attribute to each resulting {@code Element}.
106    */
107   @Override
108   protected void process(List<Element> results) throws ActionException {
109     for (Element child : results) {
110 
111       if (namespace != null) {
112         Namespace ns = NamespaceUtil.lookupOrUseGeneratedNamespace(child, getNamespace(), true);
113         if (child.getAttribute(getName(), ns) != null) {
114           String msg = "An attribute with the name '" + getName() + "' already exists.";
115           throw new ActionProcessingException(msg,
116               new IllegalArgumentException("modify-attribute should be used instead."));
117         }
118         child.setAttribute(getName(), getValue(), ns);
119       } else {
120         if (child.getAttribute(getName()) != null) {
121           String msg = "An attribute with the name '" + getName() + "' already exists.";
122           throw new ActionProcessingException(msg,
123               new IllegalArgumentException("modify-attribute should be used instead."));
124         }
125         child.setAttribute(getName(), getValue());
126       }
127     }
128 
129   }
130 }
