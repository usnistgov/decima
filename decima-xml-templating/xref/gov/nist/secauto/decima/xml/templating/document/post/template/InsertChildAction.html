27  package gov.nist.secauto.decima.xml.templating.document.post.template;
28  
29  import gov.nist.secauto.decima.core.util.ObjectUtil;
30  
31  import org.jdom2.Content;
32  import org.jdom2.Content.CType;
33  import org.jdom2.Element;
34  import org.jdom2.Namespace;
35  import org.jdom2.filter.Filters;
36  import org.jdom2.xpath.XPathFactory;
37  
38  import java.util.List;
39  import java.util.Map;
40  
41  /**
42   * Inserts new child elements within existing elements returned by an XPath query over an XML
43   * document.
44   */
45  public class InsertChildAction
46      extends AbstractXPathAction<Element> {
47  
48    private final List<Element> contentNodes;
49    private final Integer index;
50    private boolean ignoreWhitespace = true;
51  
52    /**
53     * Construct a new InsertChildAction based on an XPath string using the provided map to map XML
54     * prefixes to namespaces within the XPath.
55     * 
56     * @param xpathFactory
57     *          the XPath implementation to use
58     * @param xpath
59     *          the XPath string
60     * @param prefixToNamespaceMap
61     *          a map of XML prefixes to namespaces used in the provided XPath
62     * @param contentNodes
63     *          a list of new elements to insert
64     * @param index
65     *          for each element returned by the XPath query, the index position (zero based) identifies
66     *          the position within the element's content that the new element is to be inserted into. A
67     *          {@code null} value will cause the inserted elements to be appended at the end of the
68     *          sequence.
69     */
70    public InsertChildAction(XPathFactory xpathFactory, String xpath, Map<String, String> prefixToNamespaceMap,
71        List<Element> contentNodes, Integer index) {
72      super(xpathFactory, xpath, Filters.element(), prefixToNamespaceMap);
73      ObjectUtil.requireNonEmpty(contentNodes);
74      if (index != null && index < 0) {
75        throw new IndexOutOfBoundsException("index value '" + index + "' is not >= 0");
76      }
77      this.contentNodes = contentNodes;
78      this.index = index;
79    }
80  
81    public boolean isIgnoreWhitespace() {
82      return ignoreWhitespace;
83    }
84  
85    public void setIgnoreWhitespace(boolean ignoreWhitespace) {
86      this.ignoreWhitespace = ignoreWhitespace;
87    }
88  
89    /**
90     * Retrieves the elements to insert.
91     * 
92     * @return a list of elements
93     */
94    public List<Element> getContentNodes() {
95      return contentNodes;
96    }
97  
98    /**
99     * Retrieves the zero-based index position for which new nodes will be inserted.
100    * 
101    * @return a zero-based index position or {@code null} if new elements should be appended
102    */
103   public Integer getIndex() {
104     return index;
105   }
106 
107   /**
108    * Inserts the child elements based on the provided XPath results.
109    */
110   @Override
111   protected void process(List<Element> results) throws ActionException {
112     for (Element parent : results) {
113       Integer localIndex = getIndex();
114       if (localIndex != null) {
115         int childCount = parent.getContent().size();
116 
117         int pos = 0;
118         for (Content content : parent.getContent()) {
119           if (!CType.Element.equals(content.getCType())) {
120             ++localIndex;
121           }
122           if (pos == localIndex) {
123             break;
124           }
125           pos++;
126         }
127 
128         if (localIndex == childCount) {
129           int children = isIgnoreWhitespace() ? parent.getChildren().size() : parent.getContent().size();
130           String msg = "Index '" + getIndex() + "' is equal to the number of child elements '" + children
131               + ". Do not specify an index in this case.";
132           throw new ActionProcessingException(msg,
133               new IllegalArgumentException("index should be null instead of " + getIndex()));
134         } else if (localIndex > childCount) {
135           int children = isIgnoreWhitespace() ? parent.getChildren().size() : parent.getContent().size();
136           String msg = "Index '" + getIndex() + "' is greater than the number of child elements '" + children;
137           throw new ActionProcessingException(msg, new IndexOutOfBoundsException(msg));
138         }
139       }
140 
141       for (Element newChild : getContentNodes()) {
142         newChild = newChild.clone();
143 
144         if (localIndex == null) {
145           parent.addContent(newChild);
146         } else {
147           parent.addContent(localIndex++, newChild);
148         }
149 
150         Namespace childNs = newChild.getNamespace();
151         Namespace parentNs = parent.getNamespace();
152         if (childNs.getURI().equals(parentNs.getURI()) && !childNs.getPrefix().equals(parentNs.getPrefix())) {
153           newChild.setNamespace(parentNs);
154         }
155       }
156     }
157   }
