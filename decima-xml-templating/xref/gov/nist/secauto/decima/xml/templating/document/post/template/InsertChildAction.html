27  package gov.nist.secauto.decima.xml.templating.document.post.template;
28  
29  import gov.nist.secauto.decima.core.util.ObjectUtil;
30  
31  import org.jdom2.Content;
32  import org.jdom2.Content.CType;
33  import org.jdom2.Element;
34  import org.jdom2.Namespace;
35  import org.jdom2.filter.Filters;
36  import org.jdom2.xpath.XPathFactory;
37  
38  import java.util.List;
39  import java.util.Map;
40  
41  /**
42   * Inserts new child elements within existing elements returned by an XPath query over an XML
43   * document.
44   */
45  public class InsertChildAction extends AbstractXPathAction<Element> {
46  
47    private final List<Element> contentNodes;
48    private final Integer index;
49    private boolean ignoreWhitespace = true;
50  
51    /**
52     * Construct a new InsertChildAction based on an XPath string using the provided map to map XML
53     * prefixes to namespaces within the XPath.
54     * 
55     * @param xpathFactory
56     *          the XPath implementation to use
57     * @param xpath
58     *          the XPath string
59     * @param prefixToNamespaceMap
60     *          a map of XML prefixes to namespaces used in the provided XPath
61     * @param contentNodes
62     *          a list of new elements to insert
63     * @param index
64     *          for each element returned by the XPath query, the index position (zero based) identifies
65     *          the position within the element's content that the new element is to be inserted into. A
66     *          {@code null} value will cause the inserted elements to be appended at the end of the
67     *          sequence.
68     */
69    public InsertChildAction(XPathFactory xpathFactory, String xpath, Map<String, String> prefixToNamespaceMap,
70        List<Element> contentNodes, Integer index) {
71      super(xpathFactory, xpath, Filters.element(), prefixToNamespaceMap);
72      ObjectUtil.requireNonEmpty(contentNodes);
73      if (index != null && index < 0) {
74        throw new IndexOutOfBoundsException("index value '" + index + "' is not >= 0");
75      }
76      this.contentNodes = contentNodes;
77      this.index = index;
78    }
79  
80    public boolean isIgnoreWhitespace() {
81      return ignoreWhitespace;
82    }
83  
84    public void setIgnoreWhitespace(boolean ignoreWhitespace) {
85      this.ignoreWhitespace = ignoreWhitespace;
86    }
87  
88    /**
89     * Retrieves the elements to insert.
90     * 
91     * @return a list of elements
92     */
93    public List<Element> getContentNodes() {
94      return contentNodes;
95    }
96  
97    /**
98     * Retrieves the zero-based index position for which new nodes will be inserted.
99     * 
100    * @return a zero-based index position or {@code null} if new elements should be appended
101    */
102   public Integer getIndex() {
103     return index;
104   }
105 
106   /**
107    * Inserts the child elements based on the provided XPath results.
108    */
109   @Override
110   protected void process(List<Element> results) throws ActionException {
111     for (Element parent : results) {
112       Integer localIndex = getIndex();
113       if (localIndex != null) {
114         int childCount = parent.getContent().size();
115 
116         int pos = 0;
117         for (Content content : parent.getContent()) {
118           if (!CType.Element.equals(content.getCType())) {
119             ++localIndex;
120           }
121           if (pos == localIndex) {
122             break;
123           }
124           pos++;
125         }
126 
127         if (localIndex == childCount) {
128           int children = isIgnoreWhitespace() ? parent.getChildren().size() : parent.getContent().size();
129           String msg = "Index '" + getIndex() + "' is equal to the number of child elements '" + children
130               + ". Do not specify an index in this case.";
131           throw new ActionProcessingException(msg,
132               new IllegalArgumentException("index should be null instead of " + getIndex()));
133         } else if (localIndex > childCount) {
134           int children = isIgnoreWhitespace() ? parent.getChildren().size() : parent.getContent().size();
135           String msg = "Index '" + getIndex() + "' is greater than the number of child elements '" + children;
136           throw new ActionProcessingException(msg, new IndexOutOfBoundsException(msg));
137         }
138       }
139 
140       for (Element newChild : getContentNodes()) {
141         newChild = newChild.clone();
142 
143         if (localIndex == null) {
144           parent.addContent(newChild);
145         } else {
146           parent.addContent(localIndex++, newChild);
147         }
148 
149         Namespace childNs = newChild.getNamespace();
150         Namespace parentNs = parent.getNamespace();
151         if (childNs.getURI().equals(parentNs.getURI()) && !childNs.getPrefix().equals(parentNs.getPrefix())) {
152           newChild.setNamespace(parentNs);
153         }
154       }
155     }
156   }
