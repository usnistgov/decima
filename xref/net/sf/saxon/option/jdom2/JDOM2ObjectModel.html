189     /**
190      * Convert a Java object to an XPath value. If the supplied object is recognized as a representation
191      * of a value using this object model, the object model should convert the value to an XPath value
192      * and return this as the result. If not, it should return null. If the object is recognized but cannot
193      * be converted, an exception should be thrown
194      *
195      * @param object the object to be converted . If this is not a JDOM node, the method returns null
196      * @param config the Saxon Configuration
197      * @return either an XPath node that wraps the supplied JDOM node, or null
198      * @throws XPathException if conversion fails
199      */
200 
201     /*@Nullable*/
202     private Sequence convertObjectToXPathValue(Object object, Configuration config) throws XPathException {
203         if (isRecognizedNode(object)) {
204             if (object instanceof Document) {
205                 return wrapDocument(object, config).getRootNode();
206             } else {
207                 Document root = getDocumentRoot(object);
208                 TreeInfo docInfo = wrapDocument(root, config);
209                 return wrapNode(docInfo, object);
210             }
211         } else {
212             return null;
213         }
214     }
215 
216     /**
217      * Convert an XPath value to an object in this object model. If the supplied value can be converted
218      * to an object in this model, of the specified class, then the conversion should be done and the
219      * resulting object returned. If the value cannot be converted, the method should return null. Note
220      * that the supplied class might be a List, in which case the method should inspect the contents of the
221      * Value to see whether they belong to this object model.
222      *
223      * @param value       the XPath value to be converted
224      * @param targetClass the Java class required for the result of the conversion
225      * @return the object that results from conversion if conversion is possible, or null otherwise
226      */
227 
228     /*@Nullable*/
229     private Object convertXPathValueToObject(Sequence value, Object targetClass) {
230         Class target = (Class) targetClass;
231         if (value instanceof VirtualNode) {
232             Object u = ((VirtualNode) value).getRealNode();
233             if (target.isAssignableFrom(u.getClass())) {
234                 return u;
235             }
236         }
237         return null;
238     }

269     /**
270      * Get the document root
271      *
272      * @param node a JDOM node
273      * @return the root of the JDOM tree containing the given node
274      */
275 
276     private static Document getDocumentRoot(Object node) {
277         while (!(node instanceof Document)) {
278             if (node instanceof Element) {
279                 if (((Element) node).isRootElement()) {
280                     return ((Element) node).getDocument();
281                 } else {
282                     node = ((Element) node).getParent();
283                 }
284             } else if (node instanceof Text) {
285                 node = ((Text) node).getParent();
286             } else if (node instanceof Comment) {
287                 node = ((Comment) node).getParent();
288             } else if (node instanceof ProcessingInstruction) {
289                 node = ((ProcessingInstruction) node).getParent();
290             } else if (node instanceof Attribute) {
291                 node = ((Attribute) node).getParent();
292             } else if (node instanceof Namespace) {
293                 throw new UnsupportedOperationException("Cannot find parent of JDOM namespace node");
294             } else {
295                 throw new IllegalStateException("Unknown JDOM node type " + node.getClass());
296             }
297         }
298         return (Document) node;
299     }
