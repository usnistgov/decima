130   private MutableXMLDocument loadURLInternal(URL location) throws DocumentException {
131     log.debug("Loading: " + location.toString());
132     MutableXMLDocument retval = getCachingStrategy().retrieve(getSystemId(location));
133     if (retval == null) {
134 
135       String systemId = getSystemId(location);
136       Document document;
137       try (InputStream is = location.openStream()) {
138         SAXEngine saxEngine = getSAXEngine();
139         document = saxEngine.build(is, systemId);
140       } catch (IOException | JDOMException e) {
141         throw new DocumentException(e.getLocalizedMessage(), e);
142       }
143 
144       MutableXMLDocument xmlDocument = newJDOMDocument(document, location);
145       retval = postProcessXMLDocument(xmlDocument, new CyclePreventingDocumentResolver());
146 
147       getCachingStrategy().store(retval);
148     }
149     return retval;
150   }

152   private MutableXMLDocument loadFileInternal(File file, URL originalLocation)
153       throws FileNotFoundException, DocumentException {
154     if (originalLocation == null) {
155       try {
156         originalLocation = file.toURI().toURL();
157       } catch (MalformedURLException e) {
158         throw new DocumentException(e.getMessage(), e);
159       }
160     }
161 
162     MutableXMLDocument xmlDocument = newJDOMDocument(file, originalLocation);
163     return postProcessXMLDocument(xmlDocument, new CyclePreventingDocumentResolver());
164   }

173   @Override
174   public XMLDocument load(URL location, File destinationFile) throws DocumentException {
175     Objects.requireNonNull(location, "The location argument must be non-null");
176     Objects.requireNonNull(destinationFile, "The destinationFile argument must be non-null");
177 
178     MutableXMLDocument retval = getCachingStrategy().retrieve(getSystemId(location));
179     if (retval == null) {
180       // Make a local copy
181       try (InputStream is = location.openStream()) {
182         Files.copy(is, destinationFile.toPath());
183       } catch (IOException e) {
184         throw new DocumentException(e.getLocalizedMessage(), e);
185       }
186 
187       // Handle as a local file
188       try {
189         retval = loadFileInternal(destinationFile, location);
190       } catch (FileNotFoundException e) {
191         throw new DocumentException(e.getLocalizedMessage(), e);
192       }
193 
194       getCachingStrategy().store(retval);
195     }
196     return retval;
197   }

225   public class CyclePreventingDocumentResolver implements ResourceResolver<MutableXMLDocument> {
226     private final Deque<String> visitedUrls;
227 
228     public CyclePreventingDocumentResolver() {
229       this(new LinkedList<>());
230     }
231 
232     public CyclePreventingDocumentResolver(Deque<String> visitedUrls) {
233       this.visitedUrls = visitedUrls;
234     }
235 
236     @Override
237     public MutableXMLDocument resolve(URL url) throws DocumentException {
238       String uri = url.toString();
239       if (visitedUrls.contains(uri)) {
240         throw new DocumentException("Loading url '" + url
241             + "' would cause a document loop. The previous url loaded was: " + visitedUrls.peek());
242       }
243       visitedUrls.push(uri);
244       MutableXMLDocument retval = loadURLInternal(url);
245       if (!uri.equals(visitedUrls.pop())) {
246         throw new DocumentException("Tracking URLs for cycles has become inconsistant.");
247       }
248 
249       return retval;
250     }
251   }
252 
253   @Override
254   public MutableXMLDocument resolve(URL url) throws DocumentException {
255     return loadURLInternal(url);
