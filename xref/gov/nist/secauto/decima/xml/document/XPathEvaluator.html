27  package gov.nist.secauto.decima.xml.document;
28  
29  import gov.nist.secauto.decima.core.document.Context;
30  import gov.nist.secauto.decima.xml.assessment.result.XPathContext;
31  
32  import org.jdom2.Attribute;
33  import org.jdom2.Content;
34  import org.jdom2.filter.Filter;
35  import org.jdom2.filter.Filters;
36  
37  import java.util.List;
38  
39  import javax.xml.namespace.NamespaceContext;
40  import javax.xml.namespace.QName;
41  import javax.xml.xpath.XPathConstants;
42  import javax.xml.xpath.XPathExpressionException;
43  import javax.xml.xpath.XPathVariableResolver;
44  
45  /**
46   * Implementations of this interface provide XPath evaluation capabilities over a specific XML
47   * document.
48   */
49  public interface XPathEvaluator {
50    /**
51     * Retrieves an object that can be used for processing namespace contexts.
52     * 
53     * @return the current namespace context handler in effect, or <code>null</code> if no handler is in
54     *         effect.
55     */
56    NamespaceContext getNamespaceContext();
57  
58    /**
59     * Used to set handler for prefix-to-namespace mapping and other related operations. The utility
60     * class {@link XPathNamespaceContext} must be used to define namespace mappings.
61     * 
62     * @param nsContext
63     *          the new namespace context to use that will replace the existing context handler
64     * @throws NullPointerException
65     *           If <code>nsContext</code> is <code>null</code>.
66     */
67    void setNamespaceContext(XPathNamespaceContext nsContext);
68  
69    /**
70     * Retrieves the resolver used to process XPath variables.
71     * 
72     * @return the current variable resolver in effect, or <code>null</code> if no resolver is in
73     *         effect.
74     */
75    XPathVariableResolver getXPathVariableResolver();
76  
77    /**
78     * Set a user provided XPath variable resolver that can be used to resolve variables that appear in
79     * XPath statements.
80     * 
81     * @param resolver
82     *          the new variable resolve to use that will replace the existing resolver
83     * @throws NullPointerException
84     *           If <code>resolver</code> is <code>null</code>.
85     */
86    void setXPathVariableResolver(XPathVariableResolver resolver);
87  
88    /**
89     * Evaluates an XPath expression, returning a list of node results matching the provided filter. A
90     * filter can be created using the {@link Filters} class.
91     * 
92     * @param <T>
93     *          the type of object to be filtered against when building the result set
94     * @param xpath
95     *          the XPath expression to evaluate
96     * @param filter
97     *          the filter to use to limit the returned results, or <code>null</code> if no filter is to
98     *          be applied
99     * @return a non-null result
100    * @throws XPathExpressionException
101    *           if an error occurred while evaluating the XPath expression
102    */
103   <T> List<T> evaluate(String xpath, Filter<T> filter) throws XPathExpressionException;
104 
105   /**
106    * Evaluates an XPath expression, returning a list of node results matching the provided filter. A
107    * filter can be created using the {@link Filters} class.
108    * 
109    * @param <T>
110    *          the type of object to be filtered against when building the result set
111    * @param xpath
112    *          the XPath expression to evaluate
113    * @param returnType
114    *          the expected object type of the return value, which is one of {@link XPathConstants}
115    * @param filter
116    *          the filter to use to limit the returned results, or <code>null</code> if no filter is to
117    *          be applied
118    * @return a non-null result
119    * @throws XPathExpressionException
120    *           if an error occurred while evaluating the XPath expression
121    */
122   <T> List<T> evaluate(String xpath, QName returnType, Filter<T> filter) throws XPathExpressionException;
123 
124   /**
125    * Evaluates an XPath expression, returning a single node result matching the provided filter. A
126    * filter can be created using the {@link Filters} class.
127    * 
128    * @param <T>
129    *          the type of object to be filtered against when determining the result
130    * @param xpath
131    *          the XPath expression to evaluate
132    * @param filter
133    *          the filter to use to limit the returned result, or <code>null</code> if no filter is to
134    *          be applied
135    * @return the matching result, or <code>null</code> if no result matched
136    * @throws XPathExpressionException
137    *           if an error occurred while evaluating the XPath expression
138    */
139   <T> T evaluateSingle(String xpath, Filter<T> filter) throws XPathExpressionException;
140 
141   /**
142    * Evaluates an XPath expression, returning a single node result matching the provided filter. A
143    * filter can be created using the {@link Filters} class.
144    * 
145    * @param <T>
146    *          the type of object to be filtered against when determining the result
147    * @param xpath
148    *          the XPath expression to evaluate
149    * @param returnType
150    *          the expected object type of the return value, which is one of {@link XPathConstants}
151    * @param filter
152    *          the filter to use to limit the returned result, or <code>null</code> if no filter is to
153    *          be applied
154    * @return the matching result, or <code>null</code> if no result matched
155    * @throws XPathExpressionException
156    *           if an error occurred while evaluating the XPath expression
157    */
158   <T> T evaluateSingle(String xpath, QName returnType, Filter<T> filter) throws XPathExpressionException;
159 
160   /**
161    * Evaluates an XPath expression, returning a boolean result indicating if a non-empty result was
162    * found matching the XPath expression.
163    * 
164    * @param xpath
165    *          the XPath expression to evaluate
166    * @return <code>true</code> if at least one match to the expression was found, or
167    *         <code>false</code> otherwise
168    * @throws XPathExpressionException
169    *           if an error occurred while evaluating the XPath expression
170    */
171   boolean test(String xpath) throws XPathExpressionException;
172 
173   /**
174    * The method will return the {@link Context}, the location within an XML document, for a provided
175    * XPath expression. The provided XPath expression must evaluate to a single {@link Content} or
176    * {@link Attribute} result.
177    * 
178    * @param xpath
179    *          the XPath expression to find the context for
180    * @return the context of the evaluated XPath expression result
181    * @throws XPathExpressionException
182    *           if an error occurred while evaluating the XPath expression
183    */
184   XPathContext getContext(String xpath) throws XPathExpressionException;
185 }
