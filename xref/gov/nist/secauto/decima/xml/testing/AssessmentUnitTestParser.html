170     Document document;

174       throw new ParserException("unable to load unit test document", e);
175     }
176     return buildUnitTest(document, url);

189   protected DefaultAssessmentUnitTest buildUnitTest(Document doc, URL testURL) throws ParserException {
190     Element cut = doc.getRootElement();
191 
192     AssessmentUnitTestBuildermentUnitTestBuilder.html#AssessmentUnitTestBuilder">AssessmentUnitTestBuilder builder = new AssessmentUnitTestBuilder(getXMLDocumentFactory());
193     builder.setSourceURI(testURL.toString());
194     builder.setDerivedRequirement(cut.getChild("requirement", cut.getNamespace()).getTextNormalize());
195     builder.setSummary(cut.getChild("description", cut.getNamespace()).getTextNormalize());
196     builder.setResultDirectory(resultDir);
197 
198     TemplateProcessor tp;
199     try {
200       tp = TemplateParser.getInstance().parse(cut.getChild("template", TemplateParser.TEMPLATE_NAMESPACE), testURL);
201     } catch (TemplateParserException e) {
202       throw new ParserException("Unable to parse template", e);
203     }
204 
205     builder.setTemplate(tp);
206     Element assessmentsElement = cut.getChild("assessments", cut.getNamespace());
207     List<Element> assessments = assessmentsElement.getChildren();
208     for (int index = 0; index < assessments.size(); index++) {
209       Element assessmentElement = assessments.get(index);
210       File resultDir = new File(getResultDirectory(), "assessment-" + index);
211 
212       Assessment<XMLDocument> assessment;
213       if ("schematron-assessment".equals(assessmentElement.getName())) {
214         assessment = buildSchematronAssessment(assessmentElement, resultDir);
215       } else if ("schema-assessment".equals(assessmentElement.getName())) {
216         assessment = buildSchemaAssessment(assessmentElement);
217       } else if ("logic-assessment".equals(assessmentElement.getName())) {
218         try {
219           assessment = buildLogicAssessment(assessmentElement);
220         } catch (ClassNotFoundException | NoSuchMethodException | SecurityException | InstantiationException
221             | IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
222           throw new ParserException(e);
223         }
224       } else {
225         throw new ParserException(new UnsupportedOperationException("Unsupported assessment type: {"
226             + assessmentElement.getNamespaceURI() + "}" + assessmentElement.getName()));
227       }
228 
229       Attribute testAttr = assessmentElement.getAttribute("test");
230       if (testAttr != null) {
231         assessment = Decima.newConditionalAssessment(assessment, new XPathCondition(testAttr.getValue()));
232       }
233       builder.addAssessment(assessment);
234     }
235 
236     Element assertionsElement = cut.getChild("assertions", cut.getNamespace());
237     Element overallElement = assertionsElement.getChild("assert-overall", cut.getNamespace());
238     if (overallElement != null) {
239       builder.addAssertion(buildAssertOverall(overallElement));
240     } else {
241       for (Element assertionElement : assertionsElement.getChildren()) {
242         Assertion assertion;
243         switch (assertionElement.getName()) {
244         case "assert-derived-requirement":
245           assertion = buildAssertDerivedRequirement(assertionElement);
246           break;
247         case "assert-base-requirement":
248           assertion = buildAssertBaseRequirement(assertionElement);
249           break;
250         case "assert-remaining":
251           assertion = buildAssertRemaining(assertionElement);
252           break;
253         default:
254           throw new UnsupportedOperationException(
255               "Unsupported assertion type: {" + assertionElement.getNamespaceURI() + "}" + assertionElement.getName());
256         }
257 
258         builder.addAssertion(assertion);
259       }
260     }
261     return builder.build();
262   }

264   protected static LogicAssessment<XMLDocument> buildLogicAssessment(Element assessmentElement)
265       throws ClassNotFoundException, NoSuchMethodException, SecurityException, InstantiationException,
266       IllegalAccessException, IllegalArgumentException, InvocationTargetException, ParserException {
267     String className = assessmentElement.getAttributeValue("class");
268     List<Element> paramElements = assessmentElement.getChildren("parameter", assessmentElement.getNamespace());
269 
270     Class<?> clazz = Class.forName(className);
271     if (!LogicAssessment.class.isAssignableFrom(clazz)) {
272       throw new ParserException(
273           "Class '" + clazz.getName() + "' does not implement " + LogicAssessment.class.getName());
274     }
275 
276     Constructor<?> ctor;
277     Object[] args;
278     if (paramElements.isEmpty()) {
279       ctor = clazz.getConstructor();
280       args = null;
281     } else {
282       Class<?>[] paramTypes = new Class<?>[paramElements.size()];
283       args = new Object[paramElements.size()];
284       for (int i = 0; i < paramElements.size(); i++) {
285         paramTypes[i] = String.class;
286         args[i] = paramElements.get(i).getValue();
287       }
288       ctor = clazz.getConstructor(paramTypes);
289     }
290 
291     @SuppressWarnings("unchecked")
292     LogicAssessment<XMLDocument> retval = (LogicAssessment<XMLDocument>) ctor.newInstance(args);
293 
294     return retval;
295   }

324   protected final SchematronHandler newSchematronHandler(String handlerClassName, Schematron schematron, String phase)
325       throws ParserException {
326     // Use a specialized SchematronHandler?
327     SchematronHandler schematronHandler;
328     if (handlerClassName == null) {
329       // We now need a SchematronHandler instance
330       schematronHandler = new DefaultSchematronHandler(schematron);
331     } else {
332       // get the handler class to use
333       Class<? extends SchematronHandler> handlerClass;
334       try {
335         @SuppressWarnings("unchecked")
336         Class<? extends SchematronHandler> clazz = (Class<? extends SchematronHandler>) Class.forName(handlerClassName);
337         handlerClass = clazz;
338       } catch (ClassNotFoundException ex) {
339         throw new ParserException(ex);
340       }
341 
342       // construct the handler
343       // analyze the constructor arguments
344       // A typical constructor will allow for a Schematron, and optionally a phase (String)
345       Constructor<?> constructor = null;
346       Object[] params = null;
347       try {
348         constructor = handlerClass.getConstructor(TWO_ARGUMENT);
349         params = new Object[] { schematron, phase };
350       } catch (NoSuchMethodException ex) {
351         // do nothing
352       }
353 
354       if (constructor == null) {
355         try {
356           constructor = handlerClass.getConstructor(ONE_ARGUMENT);
357           params = new Object[] { schematron };
358         } catch (NoSuchMethodException ex) {
359           // do nothing
360         }
361       }
362 
363       if (constructor == null) {
364         try {
365           constructor = handlerClass.getConstructor();
366           params = new Object[] {};
367         } catch (NoSuchMethodException ex) {
368           throw new ParserException("the SchematronEvaluator must have a constructor with the signature of: "
369               + "Schematron, String; Schematron; or no arguments");
370         }
371       }
372 
373       // now construct the handler
374       try {
375         schematronHandler = (SchematronHandler) constructor.newInstance(params);
376       } catch (InstantiationException | IllegalAccessException | IllegalArgumentException | InvocationTargetException
377           | SecurityException e) {
378         throw new ParserException(e);
379       }
380 
381     }
382     return schematronHandler;
383   }
