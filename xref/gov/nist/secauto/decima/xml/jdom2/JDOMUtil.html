50  public class JDOMUtil {
51    private static final Format DEFAULT_FORMAT = Format.getPrettyFormat();
52  
53    public static String toString(Document doc) {
54      return toString(doc, DEFAULT_FORMAT);
55    }
56  
57    /**
58     * A convenience method that produces a String containing a JDOM2 {@link Document} that has been
59     * formatted using the provided format argument.
60     * 
61     * @param doc
62     *          the document to write to the String
63     * @param format
64     *          the formatter to use
65     * @return a string representation of the document
66     */
67    public static String toString(Document doc, Format format) {
68      XMLOutputter out = new XMLOutputter(format);
69      return out.outputString(doc);
70    }
71  
72    public static String toString(Element element) {
73      return toString(element, DEFAULT_FORMAT);
74    }
75  
76    public static String toString(Element element, Format format) {
77      XMLOutputter out = new XMLOutputter(format);
78      return out.outputString(element);
79    }
80  
81    /**
82     * A convenience method to construct a new validating {@link SAXEngine} based on a collection of
83     * schema sources. Any registered resolver extensions are also setup with the SAXEngine.
84     * 
85     * @param schemaSources
86     *          an array of schema to use for validation
87     * @return a new {@link SAXEngine} instance
88     * @throws SAXException
89     *           if an error occurred while parsing the schema
90     * @throws JDOMException
91     *           if an error occurred while constructing the {@link SAXEngine}
92     * @see ResourceResolverExtensionService
93     */
94    public static SAXEngine newValidatingSAXEngine(Source[] schemaSources) throws SAXException, JDOMException {
95      SchemaFactory schemafac = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
96      schemafac.setResourceResolver(ResourceResolverExtensionService.getInstance().getLSResolver());
97      // Schema schema = schemafac.newSchema();
98      Schema schema = schemafac.newSchema(schemaSources);
99      XMLReaderJDOMFactory factory = new XMLReaderSchemaFactory(schema);
100     SAXBuilder sax = new SAXBuilder(factory);
101     sax.setEntityResolver(ResourceResolverExtensionService.getInstance().getEntityResolver());
102     return sax.buildEngine();
103   }
104 
105   public static SAXEngine newValidatingSAXEngine(URL schema) throws SAXException, JDOMException {
106     return newValidatingSAXEngine(new Source[] { new StreamSource(schema.toExternalForm()) });
107   }
108 
109   private JDOMUtil() {
110     // disable construction
111   }
