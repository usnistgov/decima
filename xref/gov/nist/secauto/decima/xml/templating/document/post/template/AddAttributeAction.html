27  package gov.nist.secauto.decima.xml.templating.document.post.template;
28  
29  import gov.nist.secauto.decima.core.util.ObjectUtil;
30  import gov.nist.secauto.decima.xml.jdom2.NamespaceUtil;
31  
32  import org.jdom2.Element;
33  import org.jdom2.Namespace;
34  import org.jdom2.filter.Filters;
35  import org.jdom2.xpath.XPathFactory;
36  
37  import java.util.List;
38  import java.util.Map;
39  import java.util.Objects;
40  
41  /**
42   * Adds a new XML attribute to an exiting element.
43   */
44  public class AddAttributeAction
45      extends AbstractXPathAction<Element> {
46    private final String namespace;
47    private final String name;
48    private final String value;
49  
50    /**
51     * Construct a new AddAttributeAction based on an XPath string using the provided map to map XML
52     * prefixes to namespaces.
53     * 
54     * @param xpathFactory
55     *          the XPath implementation to use
56     * @param xpath
57     *          the XPath string
58     * @param prefixToNamespaceMap
59     *          a map of XML prefixes to namespaces used in the provided XPath
60     * @param namespace
61     *          the namespace of the new attribute or {@code null} if the attribute has no namespace
62     * @param name
63     *          the name of the new attribute
64     * @param value
65     *          the value of the new attrubute
66     */
67    public AddAttributeAction(XPathFactory xpathFactory, String xpath, Map<String, String> prefixToNamespaceMap,
68        String namespace, String name, String value) {
69      super(xpathFactory, xpath, Filters.element(), prefixToNamespaceMap);
70      ObjectUtil.requireNullOrNonEmpty(namespace, "namespace must be null or non-empty");
71      Objects.requireNonNull(name);
72      Objects.requireNonNull(value);
73      this.namespace = namespace;
74      this.name = name;
75      this.value = value;
76    }
77  
78    /**
79     * Retrieve the namespace URI to use for the new attribute.
80     * 
81     * @return the namespace URI
82     */
83    public String getNamespace() {
84      return namespace;
85    }
86  
87    /**
88     * Retrieve the name of the new attribute.
89     * 
90     * @return the attribute's name
91     */
92    public String getName() {
93      return name;
94    }
95  
96    /**
97     * Retrieve the new attribute's value.
98     * 
99     * @return the new attributes value
100    */
101   public String getValue() {
102     return value;
103   }
104 
105   /**
106    * Appends an attribute to each resulting {@code Element}.
107    */
108   @Override
109   protected void process(List<Element> results) throws ActionException {
110     for (Element child : results) {
111 
112       if (namespace != null) {
113         Namespace ns = NamespaceUtil.lookupOrUseGeneratedNamespace(child, getNamespace(), true);
114         if (child.getAttribute(getName(), ns) != null) {
115           String msg = "An attribute with the name '" + getName() + "' already exists.";
116           throw new ActionProcessingException(msg,
117               new IllegalArgumentException("modify-attribute should be used instead."));
118         }
119         child.setAttribute(getName(), getValue(), ns);
120       } else {
121         if (child.getAttribute(getName()) != null) {
122           String msg = "An attribute with the name '" + getName() + "' already exists.";
123           throw new ActionProcessingException(msg,
124               new IllegalArgumentException("modify-attribute should be used instead."));
125         }
126         child.setAttribute(getName(), getValue());
127       }
128     }
129 
130   }
131 }
