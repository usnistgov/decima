67    /**
68     * Writes a provided set of {@link AssessmentResults} to the provided {@link OutputStream} as an XML
69     * document.
70     * 
71     * @param results
72     *          the results of an assessment to use
73     * @param out
74     *          the {@link OutputStream} to write the XML results to
75     * @throws IOException
76     *           if an error occurs while writing to the provided {@link OutputStream}
77     */
78    public void write(AssessmentResults results, OutputStream out) throws IOException {
79      Document doc = newDocument(results);
80      XMLOutputter xout = new XMLOutputter(Format.getPrettyFormat());
81      xout.output(doc, out);
82    }

122   protected void buildProperties(Element root, Map<String, String> properties) {
123     Element element = new Element("properties", root.getNamespace());
124     root.addContent(element);
125     for (Map.Entry<String, String> entry : properties.entrySet()) {
126       Element property = new Element("property", root.getNamespace());
127       property.setAttribute("name", entry.getKey());
128       property.setText(entry.getValue());
129       element.addContent(property);
130     }
131   }
132 
133   protected void buildRequirements(Element root, AssessmentResults results) {
134     Element element = new Element("requirements", root.getNamespace());
135     root.addContent(element);
136 
137     RequirementsManager manager = results.getRequirementsManager();
138     for (URI definition : manager.getRequirementDefinitions()) {
139       Element requirement = new Element("requirement", root.getNamespace());
140       requirement.setAttribute("href", definition.toASCIIString());
141       element.addContent(requirement);
142     }
143   }
144 
145   protected void buildResults(Element root, AssessmentResults results, Map<String, String> systemIdToSubjectIdMap) {
146     Namespace namespace = root.getNamespace();
147     Element element = new Element("results", namespace);
148     root.addContent(element);
149     for (BaseRequirementResult base : results.getBaseRequirementResults()) {
150       element.addContent(buildBaseRequirement(base, namespace, systemIdToSubjectIdMap));
151     }
152   }

154   private static Element buildBaseRequirement(BaseRequirementResult result, Namespace namespace,
155       Map<String, String> systemIdToSubjectIdMap) {
156     Element retval = new Element("base-requirement", namespace);
157 
158     BaseRequirement base = result.getBaseRequirement();
159     retval.setAttribute("id", base.getId());
160 
161     Element status = new Element("status", namespace).setText(result.getStatus().name());
162     retval.addContent(status);
163 
164     for (DerivedRequirementResult derived : result.getDerivedRequirementResults()) {
165       retval.addContent(buildDerivedRequirement(derived, namespace, systemIdToSubjectIdMap));
166     }
167     return retval;
168   }
169 
170   private static Element buildDerivedRequirement(DerivedRequirementResult result, Namespace namespace,
171       Map<String, String> systemIdToSubjectIdMap) {
172     Element retval = new Element("derived-requirement", namespace);
173 
174     DerivedRequirement derived = result.getDerivedRequirement();
175     retval.setAttribute("id", derived.getId());
176 
177     Element status = new Element("status", namespace).setText(result.getStatus().name());
178     retval.addContent(status);
179     for (TestResult test : result.getTestResults()) {
180       retval.addContent(buildTestResult(test, derived, namespace, systemIdToSubjectIdMap));
181     }
182 
183     return retval;
184   }

186   private static Element buildTestResult(TestResult result, DerivedRequirement derived, Namespace namespace,
187       Map<String, String> systemIdToSubjectIdMap) {
188     Element retval = new Element("test", namespace);
189 
190     String testId = result.getTestId();
191     if (testId != null) {
192       retval.setAttribute("test-id-ref", testId);
193     }
194 
195     TestStatus status = result.getStatus();
196     retval.addContent(new Element("status", namespace).setText(status.name()));
197 
198     List<String> values = result.getResultValues();
199     String message = derived.getMessageText(values.toArray(new String[values.size()]));
200     if (message != null) {
201       retval.addContent(new Element("message", namespace).setText(message));
202     }
203 
204     Context context = result.getContext();
205     if (context != null) {
206       Element location = new Element("location", namespace).setAttribute("line", Integer.toString(context.getLine()))
207           .setAttribute("column", Integer.toString(context.getColumn()));
208 
209       if (context.getSystemId() != null) {
210         String id = systemIdToSubjectIdMap.get(context.getSystemId());
211         if (id != null) {
212           location.setAttribute("subject-ref", id);
213         } else {
214           log.error("Unable to find subject id for location systemId: {}", context.getSystemId());
215         }
216       }
217 
218       // TODO: Find a better way to handle this. Maybe a property map?
219       if (context instanceof XPathContext) {
220         String xpath = ((XPathContext) context).getXPath();
221         if (xpath != null) {
222           location.setAttribute("xpath", ((XPathContext) context).getXPath());
223         }
224       }
225       retval.addContent(location);
226     }
227 
228     return retval;
229   }

231   private static String dateToString(ZonedDateTime dateTime) {
232     return dateTime.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME);
233   }

235   /**
236    * Creates a new JDOM {@link Document} based on the provided {@link AssessmentResults}.
237    * 
238    * @param results
239    *          the results of an assessment to use
240    * @return a JDOM {@link Document} containing an XML representation of the results
241    */
242   public Document newDocument(AssessmentResults results) {
243     Element root = new Element("assessment-results", RESULT_NAMESPACE);
244 
245     root.setAttribute("start", dateToString(results.getStartTimestamp()));
246     root.setAttribute("end", dateToString(results.getEndTimestamp()));
247 
248     Map<String, String> systemIdToSubjectIdMap = buildSubjects(root, results);
249     Map<String, String> properties = results.getProperties();
250     if (!properties.isEmpty()) {
251       buildProperties(root, properties);
252     }
253     buildRequirements(root, results);
254     buildResults(root, results, systemIdToSubjectIdMap);
255     Document doc = new Document(root);
256     return doc;
257   }
