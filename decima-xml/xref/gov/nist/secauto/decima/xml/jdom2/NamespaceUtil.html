27  package gov.nist.secauto.decima.xml.jdom2;
28  
29  import org.jdom2.Element;
30  import org.jdom2.Namespace;
31  
32  import java.util.Collections;
33  import java.util.HashMap;
34  import java.util.Map;
35  
36  public class NamespaceUtil {
37  
38    private NamespaceUtil() {
39      // Disable construction
40    }
41  
42    /**
43     * Creates a map of prefix to namespace URIs based on the namespaces in scope for the provided
44     * element.
45     * 
46     * @param element
47     *          the element
48     * @return a map of prefixes to namespace URIs
49     */
50    public static Map<String, String> getPrefixToNamespaceMap(Element element) {
51      Map<String, String> retval = new HashMap<>();
52      for (Namespace ns : element.getNamespacesInScope()) {
53        retval.put(ns.getPrefix(), ns.getURI());
54      }
55      return Collections.unmodifiableMap(retval);
56    }
57  
58    /**
59     * Retrieves a namespace object for a provided element and namespace URI.
60     * 
61     * @param element
62     *          the element
63     * @param uri
64     *          the namespace URI
65     * @return the {@link Namespace} object with the URI or <code>null</code> if a matching namespace
66     *         cannot be found
67     */
68    public static Namespace lookupNamespace(Element element, String uri) {
69      Namespace retval = null;
70      for (Namespace namespace : element.getNamespacesInScope()) {
71        if (namespace.getURI().equals(uri)) {
72          retval = namespace;
73          break;
74        }
75      }
76      return retval;
77    }
78  
79    public static Namespace lookupOrUseGeneratedNamespace(Element element, String uri) {
80      return lookupOrUseGeneratedNamespace(element, uri, "ns", false);
81    }
82  
83    public static Namespace lookupOrUseGeneratedNamespace(Element element, String uri, boolean requirePrefix) {
84      return lookupOrUseGeneratedNamespace(element, uri, "ns", requirePrefix);
85    }
86  
87    /**
88     * Retrieves a namespace object for a provided element and namespace URI. If that namespace cannot
89     * be found, a new {@link Namespace} will be created with the provided defaultPrefix instead.
90     * 
91     * @param element
92     *          the element
93     * @param uri
94     *          the namespace URI
95     * @param defaultPrefix
96     *          the prefix to use if a new {@link Namespace} needs to be created
97     * @param requirePrefix
98     *          if <code>true</code> find or create a namespace that is not the default/blank namespace
99     *          prefix
100    * @return the {@link Namespace} object associated with the URI
101    */
102   public static Namespace lookupOrUseGeneratedNamespace(Element element, String uri, String defaultPrefix,
103       boolean requirePrefix) {
104     Namespace retval = lookupNamespace(element, uri);
105     if (retval == null) {
106       if (Namespace.XML_NAMESPACE.getURI().equals(uri)) {
107         retval = Namespace.XML_NAMESPACE;
108       } else {
109         retval = generateNewNamespaceWithPrefix(element, uri, defaultPrefix);
110       }
111     } else if (requirePrefix && "".equals(retval.getPrefix())) {
112       retval = generateNewNamespaceWithPrefix(element, uri, defaultPrefix);
113     }
114     return retval;
115   }
116 
117   /**
118    * Generates a new {@link Namespace} on the element for the provided URI.
119    * 
120    * @param element
121    *          the element
122    * @param uri
123    *          the namespace URI
124    * @param defaultPrefix
125    *          the prefix to use if a new {@link Namespace} needs to be created
126    * @return a new {@link Namespace} based on the provided information
127    */
128   public static Namespace generateNewNamespaceWithPrefix(Element element, String uri, String defaultPrefix) {
129     Namespace retval = null;
130     int prefixCount = 1;
131     do {
132       String prefix = defaultPrefix + prefixCount++;
133       Namespace existing = element.getNamespace(prefix);
134       if (existing == null) {
135         retval = Namespace.getNamespace(prefix, uri);
136       }
137     } while (retval == null);
138     return retval;
139   }
140 }
